\usepackage{ifluatex}

\ifluatex
    \usepackage{fontspec}
    \defaultfontfeatures[Mono]{Ligatures=NoCommon}

\else
    % pdfLaTeX - use fontenc and traditional font packages
    \usepackage[\typeencodings]{fontenc}
    % workaround at least for some symbols
    \usepackage{amssymb}
\fi
